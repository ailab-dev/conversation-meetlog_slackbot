## 3. モジュール設計

### 3.1 `app/api/slack/events/route.ts`（既存を拡張）

```typescript
export const dynamic = 'force-dynamic'

export async function POST(req: Request) {
  // ── 共通処理（フェーズ1 と同一）──
  // 1. URL Verification
  // 2. 署名検証
  const body = await req.json()
  const event = body.event

  // ── イベントタイプ分岐（フェーズ2 追加）──
  if (event.type === 'reaction_added') {
    await handleReactionAdded(event)
  } else if (event.type === 'app_mention') {
    await handleAppMention(event)
  }

  return Response.json({ ok: true })
}

// フェーズ1 の処理 + Embedding/upsert を末尾に追加
async function handleReactionAdded(event) {
  // ... フェーズ1 の既存処理 ...
  // Notion 保存完了後に追加:
  try {
    const vector = await embedText(notionData.fullText)
    await upsertVector(notionPage.id, vector, {
      title: notionData.title,
      category: notionData.category,
      channel: notionData.slackChannel,
      savedAt: notionData.savedAt,
      notionUrl: notionPage.url,
    })
  } catch (err) {
    console.error('[Vector] upsert error:', err)
    // フェーズ1 の動作には影響させない
  }
}

// フェーズ2 の RAG 処理
async function handleAppMention(event) {
  // bot_id が存在する場合はスキップ（無限ループ防止）
  if (event.bot_id) return

  // メンション部分を除いた質問テキストを抽出
  const question = event.text.replace(/<@[A-Z0-9]+>/g, '').trim()
  if (!question) {
    await postUsageHint(event.channel, event.ts)
    return
  }

  // RAG 処理
  const vector = await embedText(question)
  const results = await searchVector(vector, 5)
  const answer = await generateAnswer(question, results)
  await postAnswer(event.channel, event.ts, answer, results)
}
```

**実装メモ**

- `generateAnswer`: システムプロンプト + コンテキスト付きユーザープロンプトを構築して送信する

```typescript
// 遅延初期化パターン（フェーズ1 と統一）
function getGenAI() {
  return new GoogleGenerativeAI(process.env.GEMINI_API_KEY!)
}

export async function embedText(text: string): Promise<number[]> {
  const model = getGenAI().getGenerativeModel({ model: 'text-embedding-004' })
  const result = await model.embedContent(text)
  return result.embedding.values
}

export async function generateAnswer(
  question: string,
  context: VectorSearchResult[]
): Promise<string> {
  const model = getGenAI().getGenerativeModel({ model: 'gemini-1.5-flash' })
  const prompt = buildPrompt(question, context)
  const result = await model.generateContent(prompt)
  return result.response.text()
}
```

### 3.3 `lib/vector.ts`（新規）

```typescript
// 提供する関数

// ベクトルと メタデータを Upstash Vector に登録
export async function upsertVector(
  id: string,
  vector: number[],
  metadata: VectorMetadata
): Promise<void>

// 類似ナレッジを検索（スコア閾値: 0.75）
export async function searchVector(
  vector: number[],
  topK: number
): Promise<VectorSearchResult[]>
```

**実装メモ**

- `@upstash/vector` パッケージを使用する
- クライアントは遅延初期化（`UPSTASH_VECTOR_REST_URL` / `UPSTASH_VECTOR_REST_TOKEN`）
- `searchVector` はスコアが 0.75 未満の結果を除外して返す

```typescript
let _vectorClient: Index | null = null
function getVectorClient(): Index {
  if (!_vectorClient) {
    _vectorClient = new Index({
      url: process.env.UPSTASH_VECTOR_REST_URL!,
      token: process.env.UPSTASH_VECTOR_REST_TOKEN!,
    })
  }
  return _vectorClient
}

export async function upsertVector(
  id: string,
  vector: number[],
  metadata: VectorMetadata
): Promise<void> {
  await getVectorClient().upsert({ id, vector, metadata })
}

export async function searchVector(
  vector: number[],
  topK: number
): Promise<VectorSearchResult[]> {
  const results = await getVectorClient().query({
    vector,
    topK,
    includeMetadata: true,
  })
  return results
    .filter((r) => r.score >= 0.75)
    .map((r) => ({ score: r.score, metadata: r.metadata as VectorMetadata }))
}
```

## 5. エラーハンドリング方針

| エラー発生箇所 | 対応 |
|---------------|------|
| `embedText` 失敗（蓄積フロー） | `try/catch` でキャッチ・ログ出力のみ。フェーズ1 の処理結果は 200 OK で返す |
| `upsertVector` 失敗 | 同上 |
| `embedText` 失敗（検索フロー） | スレッドに「検索に失敗しました。」を返信し、200 OK を返す |
| `searchVector` 失敗 | 同上 |
| スコア閾値未満で結果0件 | スレッドに「関連する情報が見つかりませんでした。」を返信し、200 OK を返す |
| `generateAnswer` 失敗 | スレッドに「回答の生成に失敗しました。」を返信し、200 OK を返す |
| ボット自身の `app_mention` | `event.bot_id` を確認し、即 return（処理なし） |
